# NCUHOME JavaScript 代码风格指南

编写易理解、易交接、高可维护的代码。 🚀

# accessor-pairs

setter 必须有对应的 getter，getter 可以没有对应的 setter

// bad
const foo = {
    set bar(value) {
        return 'bar ' + value;
    }
};

// good
const foo = {
    set bar(value) {
        this.barValue = 'bar ' + value;
    },
    get bar() {
        return this.barValue;
    }
};
const bar = {
    get foo() {
        return this.fooValue;
    }
};

# array-callback-return

数组的方法除了 forEach 之外，回调函数必须有返回值

// bad
const foo = [1, 2, 3].map((num) => {
    console.log(num * num);
});

// good
const foo = [1, 2, 3].map((num) => {
    return num * num;
});

# complexity

禁止函数的循环复杂度超过 20

// bad
function foo() {
    if (i === 1) console.log(i);
    if (i === 2) console.log(i);
    if (i === 3) console.log(i);
    if (i === 4) console.log(i);
    if (i === 5) console.log(i);
    if (i === 6) console.log(i);
    if (i === 7) console.log(i);
    if (i === 8) console.log(i);
    if (i === 9) console.log(i);
    if (i === 10) console.log(i);
    if (i === 11) console.log(i);
    if (i === 12) console.log(i);
    if (i === 13) console.log(i);
    if (i === 14) console.log(i);
    if (i === 15) console.log(i);
    if (i === 16) console.log(i);
    if (i === 17) console.log(i);
    if (i === 18) console.log(i);
    if (i === 19) console.log(i);
    if (i === 20) console.log(i);
}

// good
function foo() {
    if (i === 1) console.log(i);
    if (i === 2) console.log(i);
    if (i === 3) console.log(i);
    if (i === 4) console.log(i);
    if (i === 5) console.log(i);
    if (i === 6) console.log(i);
    if (i === 7) console.log(i);
    if (i === 8) console.log(i);
    if (i === 9) console.log(i);
    if (i === 10) console.log(i);
    bar(i);
}
function bar(i) {
    if (i === 11) console.log(i);
    if (i === 12) console.log(i);
    if (i === 13) console.log(i);
    if (i === 14) console.log(i);
    if (i === 15) console.log(i);
    if (i === 16) console.log(i);
    if (i === 17) console.log(i);
    if (i === 18) console.log(i);
    if (i === 19) console.log(i);
    if (i === 20) console.log(i);
}

https://en.wikipedia.org/wiki/Cyclomatic_complexity

# constructor-super

constructor 中必须有 super

// bad
class Foo extends Bar {
    constructor() {}
}

// good
class Foo extends Bar {
    constructor() {
        super();
    }
}

# eqeqeq

必须使用 === 或 !==，禁止使用 == 或 !=

// bad
if (foo == 1) {
}
if (bar != null) {
}

// good
if (foo === 1) {
}
if (bar !== null) {
}

# for-direction

禁止方向错误的 for 循环

// bad
for (let i = 0; i &lt; 10; i--) {
    // do something
}

// good
for (let i = 0; i &lt; 10; i++) {
    // do something
}

# func-name-matching

函数赋值给变量的时候，函数名必须与变量名一致

// bad
const foo = function bar() {};

// good
const foo = function() {};
const bar = function bar() {};

# getter-return

getter 必须有返回值，并且禁止返回空

// bad
const user = {
    get name() {
        // do something
    }
};
class User {
    get name() {
        return;
    }
}

// good
const user = {
    get name() {
        return 'Alex';
    }
};
class User {
    get name() {
        return this.name;
    }
}

# guard-for-in

for in 内部必须有 hasOwnProperty

// bad
for (key in foo) {
    doSomething(key);
}

// good
for (key in foo) {
    if (Object.prototype.hasOwnProperty.call(foo, key)) {
        doSomething(key);
    }
}

# max-depth

代码块嵌套的深度禁止超过 5 层

// bad
function foo() {
    if (true) {
        if (true) {
            if (true) {
                if (true) {
                    if (true) {
                        if (true) {
                        }
                    }
                }
            }
        }
    }
}

// good
function foo() {
    if (true) {
        if (true) {
            if (true) {
                if (true) {
                    if (true) {
                    }
                }
            }
        }
    }
}

# max-nested-callbacks

回调函数嵌套禁止超过 3 层，多了请用 async await 替代

// bad
foo(() => {
    bar(() => {
        baz(() => {
            qux(() => {});
        });
    });
});

// good
foo(async () => {
    await bar();
    await baz();
    await qux();
});

# max-params

函数的参数禁止超过 7 个

// bad
function foo(a1, a2, a3, a4, a5, a6, a7, a8) {}

// good
function foo(a1, a2, a3, a4, a5, a6, a7) {}
function bar({ a1, a2, a3, a4, a5, a6, a7, a8 }) {}

# new-cap

new 后面的类名必须首字母大写

// bad
new foo();
new foo.bar();

// good
new Foo();
new foo.Bar();
Foo();

# no-array-constructor

禁止使用 Array 构造函数时传入的参数超过一个

// bad
const foo = Array(0, 1, 2); // [0, 1, 2]
const bar = new Array(0, 1, 2); // [0, 1, 2]

// good
const foo = [0, 1, 2];
Array(3); // [empty × 3]
new Array(3); // [empty × 3]
Array(3).fill('foo'); // ["foo", "foo", "foo"]
new Array(3).fill('foo'); // ["foo", "foo", "foo"]

参数为一个时表示创建一个指定长度的数组，比较常用 参数为多个时表示创建一个指定内容的数组，此时可以用数组字面量实现，不必使用构造函数

# no-async-promise-executor

禁止将 async 函数做为 new Promise 的回调函数

// bad
new Promise(async (resolve) => {
    setTimeout(resolve, 1000);
});

// good
new Promise((resolve) => {
    setTimeout(resolve, 1000);
});

出现这种情况时，一般不需要使用 new Promise 实现异步了

# no-buffer-constructor

禁止直接使用 Buffer

// bad
new Buffer(5);
Buffer([1, 2, 3]);

// good
Buffer.alloc(5);
Buffer.from([1, 2, 3]);

Buffer 构造函数是已废弃的语法

# no-caller

禁止使用 caller 或 callee

// bad
function foo(n) {
    if (n &lt;= 0) {
        return;
    }
    arguments.callee(n - 1);
}

// good
function foo(n) {
    if (n &lt;= 0) {
        return;
    }
    foo(n - 1);
}

它们是已废弃的语法

# no-case-declarations

switch 的 case 内有变量定义的时候，必须使用大括号将 case 内变成一个代码块

// bad
switch (foo) {
    case 1:
        const x = 1;
        break;
}

// good
switch (foo) {
    case 1: {
        const x = 1;
        break;
    }
}

# no-class-assign

禁止对已定义的 class 重新赋值

// bad
class Foo {}
Foo = {};

// good
class Foo {}

# no-compare-neg-zero

禁止与负零进行比较

// bad
if (foo === -0) {
}

// good
if (foo === 0) {
}

# no-cond-assign

禁止在测试表达式中使用赋值语句，除非这个赋值语句被括号包起来了

// bad
if (foo = 0) {
}

// good
if (foo === 0) {
}
if (bar === (foo = 0)) {
}

# no-const-assign

禁止对使用 const 定义的常量重新赋值

// bad
const foo = 1;
foo = 2;

// good
let foo = 1;
foo = 2;

for (const bar in [1, 2, 3]) {
    console.log(bar);
}

# no-constant-condition

禁止将常量作为分支条件判断中的测试表达式，但允许作为循环条件判断中的测试表达式

// bad
if (true) {
}
const foo = 0 ? 'bar' : 'baz';

// good
for (; true; ) {
    if (foo === 0) break;
}
while (true) {
    if (foo === 0) break;
}

# no-debugger

禁止使用 debugger

// bad
if (foo) {
    debugger;
}

// good
if (foo) {
    // debugger;
}

# no-dupe-class-members

禁止重复定义类的成员

// bad
class Foo {
    bar() {}
    bar() {}
}

// good
class Foo {
    bar() {}
    baz() {}
}

# no-dupe-keys

禁止在对象字面量中出现重复的键名

// bad
const foo = {
    bar: 1,
    bar: 2
};

// good
const foo = {
    bar: 1,
    baz: 2
};

# no-duplicate-case

禁止在 switch 语句中出现重复测试表达式的 case

// bad
switch (foo) {
    case 1:
        break;
    case 2:
        break;
    case 1:
        break;
}

// good
switch (foo) {
    case 1:
        break;
    case 2:
        break;
    case 3:
        break;
}

# no-duplicate-imports

禁止重复导入模块

// bad
import { readFile } from 'fs';
import { writeFile } from 'fs';

// good
import { readFile, writeFile } from 'fs';

# no-empty

禁止出现空代码块，允许 catch 为空代码块

// bad
if (foo) {
}

// good
if (foo) {
    // do something
}
try {
    // do something
} catch (e) {}

# no-empty-character-class

禁止在正则表达式中使用空的字符集 []

// bad
const reg = /abc[]/;

// good
const reg = /abc[a-z]/;

# no-empty-pattern

禁止解构赋值中出现空 {} 或 []

// bad
const {} = foo;

// good
const { bar } = foo;

# no-eq-null

禁止使用 foo == null，必须使用 foo === null

// bad
if (foo == null) {
}

// good
if (foo === null) {
}

# no-eval

禁止使用 eval

// bad
eval('const foo = 0');

// good
const foo = 0;

# no-ex-assign

禁止将 catch 的第一个参数 error 重新赋值

// bad
try {
} catch (e) {
    e = 10;
}

// good
try {
} catch (e) {
    console.error(e);
}

# no-extend-native

禁止修改原生对象

// bad
Array.prototype.flat = function() {
    // do something
};

[1, [2, 3]].flat();

// good
function flat(arr) {
    // do something
}

flat([1, [2, 3]]);

修改原生对象可能会与将来版本的 js 冲突

# no-extra-bind

禁止出现没必要的 bind

// bad
(function() {
    foo();
}.bind(bar));

// good
(function() {
    this.foo();
}.bind(bar));

# no-extra-boolean-cast

禁止不必要的布尔类型转换

// bad
if (!!foo) {
}
if (Boolean(foo)) {
}

// good
if (foo) {
}
if (!foo) {
}

# no-fallthrough

switch 的 case 内必须有 break, return 或 throw，空的 case 除外

// bad
switch (foo) {
    case 1:
        doSomething();
    case 2:
        doSomethingElse();
}

// good
switch (foo) {
    case 1:
        doSomething();
        break;
    case 2:
        doSomethingElse();
}
switch (foo) {
    case 1:
    case 2:
        doSomething();
}

# no-func-assign

禁止将一个函数声明重新赋值

// bad
function foo() {}
foo = 1;

// good
let foo = function() {};
foo = 1;

# no-global-assign

禁止对全局变量赋值

// bad
Object = null;

// good
foo = null;

# no-implicit-coercion

禁止使用 ~+ 等难以理解的类型转换，仅允许使用 !!

// bad
const b = ~foo.indexOf('.');
const n = +foo;
const m = 1 * foo;
const s = '' + foo;
foo += '';

// good
const b = foo.indexOf('.') !== -1;
const n = Number(foo);
const m = Number(foo);
const s = String(foo);
foo = String(foo);

const c = !!foo;

# no-implied-eval

禁止在 setTimeout 或 setInterval 中传入字符串

// bad
setTimeout('alert("Hello World");', 1000);

// good
setTimeout(() => {
    alert('Hello World');
}, 1000);

# no-inner-declarations

禁止在 if 代码块内出现函数声明

// bad
if (foo) {
    function bar() {}
}

// good
if (foo) {
    const bar = function() {};
}

# no-invalid-regexp

禁止在 RegExp 构造函数中出现非法的正则表达式

// bad
const reg1 = new RegExp('[');
const reg2 = new RegExp('.', 'z');

// good
const reg1 = new RegExp('[a-z]');
const reg2 = new RegExp('.', 'g');

# no-invalid-this

禁止在类之外的地方使用 this

// bad
function foo() {
    this.a = 0;
}

// good
class Foo {
    constructor() {
        this.a = 0;
    }
}

只允许在 class 中使用 this

# no-irregular-whitespace

禁止使用特殊空白符（比如全角空格），除非是出现在字符串、正则表达式或模版字符串中

// bad
function foo()　{
}

// good
const foo = '　';
const bar = /　/;
const baz = `　`;

# no-iterator

禁止使用 iterator

// bad
Foo.prototype.__iterator__ = function() {
    return new FooIterator(this);
};

// good
let foo = {};
foo[Symbol.iterator] = function*() {
    yield 1;
    yield 2;
    yield 3;
};
console.log([...foo]);
// [1, 2, 3]

iterator 是一个已废弃的属性 使用 [Symbol.iterator] 替代它

# no-labels

禁止使用 label

// bad
loop:
    for (let i = 0; i &lt; 5; i++) {
        if (i === 1) {
            continue loop;
        }
        console.log(i);
    }
// 0 2 3 4

// good
for (let i = 0; i &lt; 5; i++) {
    if (i === 1) {
        continue;
    }
    console.log(i);
}
// 0 2 3 4

# no-lone-blocks

禁止使用没必要的 {} 作为代码块

// bad
{
    foo();
}

// good
if (foo) {
    bar();
}

# no-misleading-character-class

禁止正则表达式中使用肉眼无法区分的特殊字符

// bad
/^[Á]$/u.test('Á'); // false

// good
/^[A]$/u.test('A'); // true

某些特殊字符很难看出差异，最好不要在正则中使用

# no-multi-str

禁止使用 \ 来换行字符串

// bad
const foo = 'Line 1\
Line 2';

// good
const foo = `Line 1
Line 2`;

# no-new

禁止直接 new 一个类而不赋值

// bad
new Foo();

// good
const foo = new foo();

new 应该作为创建一个类的实例的方法，所以不能不赋值

# no-new-func

禁止使用 new Function

// bad
const foo = new Function('a', 'b', 'return a + b');

// good
const foo = function(a, b) {
    return a + b;
};

这和 eval 是等价的

# no-new-object

禁止直接 new Object

// bad
const foo = new Object();

// good
const foo = {};

# no-new-require

禁止直接 new require('foo')

// bad
const foo = new require('foo');

// good
const Foo = require('foo');
const foo = new Foo();

# no-new-symbol

禁止使用 new 来生成 Symbol

// bad
const foo = new Symbol('foo');

// good
const foo = Symbol('foo');

# no-new-wrappers

禁止使用 new 来生成 String, Number 或 Boolean

// bad
const s = new String('foo');
const n = new Number(1);
const b = new Boolean(true);

// good
const s = String(someValue);
const n = Number(someValue);
const b = Boolean(someValue);

# no-obj-calls

禁止将 Math, JSON 或 Reflect 直接作为函数调用

// bad
const foo = Math();
const bar = JSON();
const baz = Reflect();

// good
const foo = Math.random();
const bar = JSON.parse('{}');
const baz = Reflect.get({ x: 1, y: 2 }, 'x');

# no-param-reassign

禁止对函数的参数重新赋值

// bad
function foo(bar) {
    bar = bar || '';
}

// good
function foo(bar_) {
    bar = bar_ || '';
}

# no-path-concat

禁止对 __dirname 或 __filename 使用字符串连接

// bad
const foo = __dirname + '/foo.js';
const bar = __filename + '/bar.js';

// good
import path from 'path';

const foo = path.resolve(__dirname, 'foo.js');
const bar = path.join(__filename, 'bar.js');

不同平台下的路径符号不一致，建议使用 path 处理平台差异性

# no-proto

禁止使用 proto

// bad
const foo = bar.__proto__;
bar.__proto__ = baz;

// good
const foo = Object.getPrototypeOf(bar);
Object.setPrototypeOf(bar, baz);

proto 是已废弃的语法

# no-regex-spaces

禁止在正则表达式中出现连续的空格

// bad
const reg1 = /foo   bar/;
const reg2 = new RegExp('foo   bar');

// good
const reg1 = /foo {3}bar/;
const reg2 = new RegExp('foo {3}bar');

# no-return-assign

禁止在 return 语句里赋值

// bad
function foo() {
    return (bar = 1);
}

// good
function foo() {
    bar = 1;
    return bar;
}

# no-return-await

禁止在 return 语句里使用 await

// bad
async function foo() {
    return await bar();
}

// good
async function foo() {
    const b = await bar();
    return b;
}

# no-self-assign

禁止将自己赋值给自己

// bad
foo = foo;

// good
foo = bar;

# no-self-compare

禁止将自己与自己比较

// bad
if (foo === foo) {
}
if (NaN === NaN) {
}

// good
if (foo === bar) {
}
if (isNaN(foo)) {
}

# no-sequences

禁止使用逗号操作符

// bad
foo = doSomething(), 1;

// good
doSomething();
foo = 1;

# no-shadow-restricted-names

禁止使用保留字作为变量名

// bad
const undefined = 1;
function foo(NaN) {}
function Infinity() {}

// good
console.log(undefined);
console.log(NaN);
console.log(Infinity);

# no-sparse-arrays

禁止在数组中出现连续的逗号

// bad
const foo = [1, 2, , 3];

// good
const foo = [1, 2, 3];

# no-template-curly-in-string

禁止在普通字符串中出现模版字符串里的变量形式

// bad
const foo = 'Hello ${bar}';

// good
const foo = 'Hello {bar}';

# no-this-before-super

禁止在 super 被调用之前使用 this 或 super

// bad
class Foo extends Bar {
    constructor() {
        this.foo = 1;
        super();
    }
}

// good
class Foo extends Bar {
    constructor() {
        super();
        this.foo = 1;
    }
}

# no-throw-literal

禁止 throw 字面量，必须 throw 一个 Error 对象

// bad
throw 'foo';
throw 1;

// good
throw new Error('foo');

# no-undef

禁止使用未定义的变量

// bad
foo(bar);

// good
function foo() {}
const bar = 1;
foo(bar);

if (typeof baz === 'number') {
}

# no-undef-init

禁止将 undefined 赋值给变量

// bad
let foo = undefined;

// good
let foo;

# no-unmodified-loop-condition

循环内必须对循环条件中的变量有修改

// bad
let foo = 10;
while (foo) {
    console.log(foo);
}

// good
let foo = 10;
while (foo) {
    console.log(foo);
    foo--;
}

# no-unreachable

禁止在 return, throw, break 或 continue 之后还有代码

// bad
function foo() {
    return;
    const bar = 1;
}

// good
function foo() {
    return;
    // const bar = 1;
}

# no-unsafe-finally

禁止在 finally 中出现 return, throw, break 或 continue

// bad
function foo() {
    try {
        return 1;
    } finally {
        // finally 会在 try 之前执行，故会 return 2
        return 2;
    }
}

// good
function foo() {
    try {
        return 1;
    } finally {
        console.log(2);
    }
}

finally 中的语句会在 try 之前执行

# no-unsafe-negation

禁止在 in 或 instanceof 操作符的左侧变量前使用感叹号

// bad
if (!key in object) {
}
if (!obj instanceof SomeClass) {
}

// good
if (!(key in object)) {
}
if (!(obj instanceof SomeClass)) {
}

# no-unused-expressions

禁止无用的表达式

// bad
1;
foo;
('foo');
foo &amp;&amp; bar;
foo || bar;
foo ? bar : baz;
`bar`;

// good
'use strict';
foo &amp;&amp; bar();
foo || bar();
foo ? bar() : baz();
foo`bar`;

# no-unused-vars

已定义的变量必须使用

// bad
let foo = 1;
foo = 2;

function bar(baz) {}

const { baz, ...rest } = data;

// good
let foo = 1;
console.log(foo);

function bar(baz) {}
bar();

const { baz, ...rest } = data;
console.log(baz, rest);

try {
} catch (e) {}

# no-use-before-define

变量必须先定义后使用

// bad
console.log(foo);
const foo = 1;

new Baz();
class Baz {}

// good
(() => {
    console.log(foo);
})();
const foo = 1;
console.log(foo);

bar();
function bar() {}

(() => {
    new Baz();
})();
class Baz {}
new Baz();

# no-useless-call

禁止出现没必要的 call 或 apply

// bad
foo.call(null, 1, 2, 3); // foo(1, 2, 3)
foo.apply(null, [1, 2, 3]); // foo(1, 2, 3)

foo.bar.call(foo, 1, 2, 3); // foo.bar(1, 2, 3);
foo.bar.apply(foo, [1, 2, 3]); // foo.bar(1, 2, 3);

// good
foo.call(bar, 1, 2, 3);
foo.apply(bar, [1, 2, 3]);

foo.bar.call(baz, 1, 2, 3);
foo.bar.apply(baz, [1, 2, 3]);

# no-useless-catch

禁止在 catch 中仅仅只是把错误 throw 出去

// bad
try {
    doSomethingThatMightThrow();
} catch (e) {
    throw e;
}

// good
doSomethingThatMightThrow();

try {
    doSomethingThatMightThrow();
} catch (e) {
    doSomethingBeforeRethrow();
    throw e;
}

这样的 catch 是没有意义的，等价于直接执行 try 里的代码

# no-useless-computed-key

禁止出现没必要的计算键名

// bad
const foo = {
    ['1']: 1,
    ['bar']: 'bar'
};

// good
const foo = {
    1: 1,
    bar: 'bar'
};

# no-useless-concat

禁止出现没必要的字符串连接

// bad
const foo = 'f' + 'oo';
const bar = `b` + `ar`;

// good
const foo = 'fo';
const bar = 1 + `ar`;

# no-useless-constructor

禁止出现没必要的 constructor

// bad
class Foo {
    constructor() {}
}
class Bar extends Foo {
    constructor(...args) {
        super(...args);
    }
}

// good
class Foo {
    constructor() {
        doSomething();
    }
}
class Bar extends Foo {
    constructor(...args) {
        super(...args);
        doSomething();
    }
}

# no-useless-rename

禁止解构赋值时出现同样名字的的重命名，比如 let { foo: foo } = bar;

// bad
import { foo as foo } from 'foo';
const bar = 1;
export { bar as bar };
let { baz: baz } = foo;

// good
import { foo } from 'foo';
const bar = 1;
export { bar };
let { baz } = foo;

# no-var

禁止使用 var

// bad
var foo = 1;

// good
let foo = 1;
const bar = 2;

# no-void

禁止使用 void

// bad
function foo() {
    return void 0;
}

// good
function foo() {
    return;
}

# one-var

禁止变量申明时用逗号一次申明多个

// bad
let foo, bar;
const baz = 1,
    qux = 2;

// good
let foo;
let bar;
const baz = 1;
const qux = 2;

# prefer-object-spread

必须使用 ... 而不是 Object.assign，除非 Object.assign 的第一个参数是一个变量

// bad
const foo = Object.assign({}, bar);

// good
const foo = { ...bar };

// 第一个参数为变量时允许使用 Object.assign
Object.assign(foo, baz);

# prefer-promise-reject-errors

Promise 的 reject 中必须传入 Error 对象，而不是字面量

// bad
Promise.reject('foo');

new Promise((resolve, reject) => {
    reject();
});

new Promise((resolve, reject) => {
    reject('foo');
});

// good
Promise.reject(new Error('foo'));

new Promise((resolve, reject) => {
    reject(new Error('foo'));
});

# radix

parseInt 必须传入第二个参数

// bad
const foo = parseInt('071'); // 57

// good
const foo = parseInt('071', 10); // 71

# require-yield

generator 函数内必须有 yield

// bad
function* foo() {
    return 1;
}

// good
function* foo() {
    yield 1;
    return 2;
}

# spaced-comment

注释的斜线或 * 后必须有空格

// bad
//foo
/*bar */
/**baz */

// good
// foo
/* bar */
/** baz */

# strict

禁止使用 'strict';

// bad
'use strict';
function foo() {
    'use strict';
}

// good
function foo() {}

# symbol-description

创建 Symbol 时必须传入参数

// bad
const foo = Symbol();

// good
const foo = Symbol('foo');

# use-isnan

必须使用 isNaN(foo) 而不是 foo === NaN

// bad
if (foo === NaN) {
}

// good
if (isNaN(foo)) {
}

# valid-typeof

typeof 表达式比较的对象必须是 'undefined', 'object', 'boolean', 'number', 'string', 'function', 'symbol', 或 'bigint'

// bad
if (typeof foo === 'numbe') {
}

// good
if (typeof foo === 'number') {
}

# yoda

必须使用 if (foo === 5) 而不是 if (5 === foo)

// bad
if (1 === foo) {
}

// good
if (foo === 1) {
}
if (1 &lt; foo) {
}
if (1 &lt; foo &amp;&amp; foo &lt; 10) {
}
